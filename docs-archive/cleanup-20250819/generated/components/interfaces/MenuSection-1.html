export const getDaypartSection = (): string => {
  const hour = new Date().getHours();
  if (hour < 11) return 'small-plates'; // Before lunch
  if (hour < 14) return 'lunch-combos'; // Lunch time
  return 'fresh-bowls'; // After lunch
};
